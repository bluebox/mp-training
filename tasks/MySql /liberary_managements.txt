
-- creation of tables
create table Book(
  book_id int Primary Key,
  name varchar(50),
  author varchar(50),
  total_books int,
  status int,
  cost_of_book int
  );


create table Account(
  id int primary key,
  paid_amount int,
  renwal_date date,
  status varchar(50)
  );


create table Student(
  htno int primary key,
  name varchar(50),
  year int,
  branch varchar(50),
  lib_accno int,
  Foreign key (lib_accno) References Account(i),
  Foreign key (lib_accno) References Student_books(student_htno)
);

  
create table Students_books(
  id int primary key,
  student_htno int,
  book1 int,
  book2 int,
  book3 int,
  book_took_on date,
  book_return_on date,
  Foreign key (id) References Account(i),
  Foreign key (student_htno) References Student(htno),
  Foreign key (book1) References Book(book_id),
  Foreign key (book2) References Book(book_id),
  Foreign key (book3) References Book(book_id));
